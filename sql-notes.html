03 — Foundations

DDL — Create, Alter, Drop

DDL statements define and modify the structure of database objects like tables, indexes, and schemas.

SQL — CREATE TABLE
CREATE TABLE employees (
    id          INT           PRIMARY KEY AUTO_INCREMENT,
    name        VARCHAR(100)  NOT NULL,
    email       VARCHAR(255)  UNIQUE,
    department  VARCHAR(50),
    salary      DECIMAL(10,2) DEFAULT 0.00,
    hire_date   DATE          NOT NULL,
    is_active   BOOLEAN       DEFAULT TRUE
);
SQL — ALTER & DROP
-- Add a column
ALTER TABLE employees ADD COLUMN phone VARCHAR(20);

-- Modify a column type
ALTER TABLE employees MODIFY COLUMN salary DECIMAL(12,2);

-- Rename a column (PostgreSQL / MySQL 8+)
ALTER TABLE employees RENAME COLUMN phone TO phone_number;

-- Drop a column
ALTER TABLE employees DROP COLUMN phone_number;

-- Drop the table entirely
DROP TABLE employees;

-- Drop only if it exists (safe)
DROP TABLE IF EXISTS employees;

-- Remove all rows but keep structure
TRUNCATE TABLE employees;
04 — Querying

SELECT Basics

The SELECT statement retrieves data. The full clause order is: SELECT → FROM → WHERE → GROUP BY → HAVING → ORDER BY → LIMIT.

SQL — SELECT Anatomy
-- Select all columns
SELECT * FROM employees;

-- Select specific columns with alias
SELECT
    id,
    name        AS full_name,
    salary      AS annual_salary,
    department
FROM employees;

-- Computed column
SELECT
    name,
    salary,
    salary * 1.10 AS salary_with_raise
FROM employees;

-- Remove duplicate rows
SELECT DISTINCT department FROM employees;

-- Sort results
SELECT name, salary
FROM employees
ORDER BY salary DESC, name ASC;

-- Limit rows
SELECT name FROM employees ORDER BY salary DESC LIMIT 5;

-- Pagination (OFFSET)
SELECT name FROM employees
ORDER BY id
LIMIT 10 OFFSET 20; -- page 3 of 10
05 — Querying

WHERE & Filtering

SQL — WHERE Conditions
-- Comparison operators
SELECT * FROM employees WHERE salary > 50000;
SELECT * FROM employees WHERE department = 'Engineering';
SELECT * FROM employees WHERE salary != 0;

-- Logical operators
SELECT * FROM employees
WHERE department = 'Engineering' AND salary > 70000;

SELECT * FROM employees
WHERE department = 'HR' OR department = 'Finance';

SELECT * FROM employees WHERE NOT is_active;

-- BETWEEN (inclusive)
SELECT * FROM employees
WHERE salary BETWEEN 40000 AND 80000;

-- IN — match a list
SELECT * FROM employees
WHERE department IN ('Engineering', 'Design', 'Product');

-- LIKE — pattern matching
SELECT * FROM employees WHERE name LIKE 'A%';    -- starts with A
SELECT * FROM employees WHERE email LIKE '%@gmail.com';
SELECT * FROM employees WHERE name LIKE '_ohn';   -- _ = any 1 char

-- NULL checks
SELECT * FROM employees WHERE department IS NULL;
SELECT * FROM employees WHERE department IS NOT NULL;
NoteNever use = NULL. Always use IS NULL or IS NOT NULL.
06 — Querying

JOINs

JOINs combine rows from two or more tables based on a related column.

Join TypeReturns
INNER JOINRows with matching values in both tables
LEFT JOINAll rows from left + matching rows from right (NULLs if no match)
RIGHT JOINAll rows from right + matching rows from left
FULL OUTER JOINAll rows from both tables
CROSS JOINCartesian product — every combination
SELF JOINTable joined with itself
SQL — JOIN Examples
-- INNER JOIN: only matched rows
SELECT
    e.name,
    d.department_name,
    e.salary
FROM       employees   e
INNER JOIN departments d ON e.department_id = d.id;

-- LEFT JOIN: all employees, even without a department
SELECT e.name, d.department_name
FROM      employees   e
LEFT JOIN departments d ON e.department_id = d.id;

-- Find employees with NO department
SELECT e.name
FROM      employees   e
LEFT JOIN departments d ON e.department_id = d.id
WHERE d.id IS NULL;

-- SELF JOIN: employee and their manager
SELECT
    e.name       AS employee,
    m.name       AS manager
FROM      employees e
LEFT JOIN employees m ON e.manager_id = m.id;

-- Multiple JOINs
SELECT e.name, d.department_name, l.city
FROM       employees   e
INNER JOIN departments d ON e.dept_id   = d.id
INNER JOIN locations   l ON d.location_id = l.id;
07 — Querying

GROUP BY & Aggregate Functions

FunctionDescription
COUNT(*)Count all rows
COUNT(col)Count non-NULL values
SUM(col)Sum of values
AVG(col)Average value
MIN(col)Minimum value
MAX(col)Maximum value
GROUP_CONCATConcatenate values (MySQL)
STRING_AGGConcatenate values (PostgreSQL)
SQL — GROUP BY & HAVING
-- Count employees per department
SELECT
    department,
    COUNT(*)    AS headcount,
    AVG(salary)  AS avg_salary,
    MAX(salary)  AS top_salary,
    SUM(salary)  AS total_payroll
FROM employees
GROUP BY department;

-- HAVING filters AFTER grouping (WHERE filters BEFORE)
SELECT department, COUNT(*) AS headcount
FROM employees
GROUP BY department
HAVING COUNT(*) > 5
ORDER BY headcount DESC;

-- Combine WHERE + GROUP BY + HAVING
SELECT department, AVG(salary) AS avg_sal
FROM employees
WHERE is_active = TRUE
GROUP BY department
HAVING AVG(salary) > 60000
ORDER BY avg_sal DESC;
TipUse WHERE to filter rows before aggregation. Use HAVING to filter groups after aggregation.
08 — Querying

Subqueries & CTEs

SQL — Subqueries
-- Scalar subquery (returns single value)
SELECT name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

-- IN subquery
SELECT name FROM employees
WHERE department_id IN (
    SELECT id FROM departments WHERE location = 'NYC'
);

-- EXISTS — true if subquery returns any row
SELECT name FROM employees e
WHERE EXISTS (
    SELECT 1 FROM projects p
    WHERE p.lead_id = e.id
);

-- Derived table (subquery in FROM)
SELECT dept, avg_sal
FROM (
    SELECT department AS dept, AVG(salary) AS avg_sal
    FROM employees
    GROUP BY department
) dept_avg
WHERE avg_sal > 55000;
SQL — CTEs (Common Table Expressions)
-- Basic CTE with WITH clause
WITH high_earners AS (
    SELECT id, name, salary, department
    FROM employees
    WHERE salary > 90000
)
SELECT department, COUNT(*) AS top_earners
FROM high_earners
GROUP BY department;

-- Multiple CTEs
WITH
  dept_stats AS (
      SELECT department, AVG(salary) avg_sal
      FROM employees GROUP BY department
  ),
  top_depts AS (
      SELECT department FROM dept_stats WHERE avg_sal > 70000
  )
SELECT e.name, e.department
FROM employees e
INNER JOIN top_depts t ON e.department = t.department;

-- Recursive CTE (org chart / hierarchy)
WITH RECURSIVE org_chart AS (
    -- Base case: top-level (no manager)
    SELECT id, name, manager_id, 1 AS level
    FROM employees
    WHERE manager_id IS NULL

    UNION ALL

    -- Recursive case
    SELECT e.id, e.name, e.manager_id, o.level + 1
    FROM employees e
    INNER JOIN org_chart o ON e.manager_id = o.id
)
SELECT * FROM org_chart ORDER BY level, name;
09 — Querying

Window Functions

Window functions perform calculations across a set of rows without collapsing them into a single output row (unlike GROUP BY).

FunctionDescription
ROW_NUMBER()Sequential row number (no ties)
RANK()Rank with gaps on ties
DENSE_RANK()Rank without gaps on ties
NTILE(n)Divide rows into n buckets
LAG(col, n)Value from n rows before
LEAD(col, n)Value from n rows ahead
SUM() OVERRunning / partitioned sum
AVG() OVERRunning / partitioned average
FIRST_VALUE()First value in window frame
LAST_VALUE()Last value in window frame
SQL — Window Function Examples
-- Rank employees by salary within each department
SELECT
    name,
    department,
    salary,
    RANK()       OVER (PARTITION BY department ORDER BY salary DESC) AS dept_rank,
    DENSE_RANK() OVER (PARTITION BY department ORDER BY salary DESC) AS dense_rank,
    ROW_NUMBER() OVER (PARTITION BY department ORDER BY salary DESC) AS row_num
FROM employees;

-- Running total of salary
SELECT
    name,
    hire_date,
    salary,
    SUM(salary) OVER (ORDER BY hire_date) AS running_total
FROM employees;

-- LAG: compare each salary to the previous row
SELECT
    name,
    salary,
    LAG(salary, 1) OVER (ORDER BY salary) AS prev_salary,
    salary - LAG(salary, 1) OVER (ORDER BY salary) AS diff
FROM employees;

-- Top 1 employee per department
SELECT * FROM (
    SELECT name, department, salary,
           ROW_NUMBER() OVER (PARTITION BY department ORDER BY salary DESC) rn
    FROM employees
) ranked
WHERE rn = 1;
10 — Manipulation

INSERT / UPDATE / DELETE

SQL — DML Statements
-- INSERT single row
INSERT INTO employees (name, email, department, salary, hire_date)
VALUES ('Jane Smith', 'jane@company.com', 'Engineering', 85000, '2025-03-24');

-- INSERT multiple rows
INSERT INTO employees (name, department, salary, hire_date)
VALUES
    ('Alice', 'Design',      72000, '2025-01-15'),
    ('Bob',   'Marketing',   65000, '2025-02-01'),
    ('Carol', 'Engineering', 91000, '2025-03-10');

-- INSERT from SELECT
INSERT INTO archive_employees
SELECT * FROM employees WHERE is_active = FALSE;

-- UPDATE rows
UPDATE employees
SET
    salary     = salary * 1.05,
    department = 'Senior Engineering'
WHERE department = 'Engineering' AND salary > 80000;

-- DELETE rows
DELETE FROM employees WHERE is_active = FALSE;

-- UPSERT (INSERT OR UPDATE) — PostgreSQL
INSERT INTO employees (id, name, salary)
VALUES (1, 'Jane', 90000)
ON CONFLICT (id) DO UPDATE SET
    salary = EXCLUDED.salary;
WarningAlways include a WHERE clause in UPDATE and DELETE. Without it, you modify or delete every row in the table.
11 — Manipulation

Transactions

Transactions group multiple SQL statements into a single atomic unit. They follow ACID properties: Atomicity, Consistency, Isolation, Durability.

SQL — Transactions
-- Basic transaction
BEGIN;   -- or START TRANSACTION;

UPDATE accounts SET balance = balance - 500 WHERE id = 1;
UPDATE accounts SET balance = balance + 500 WHERE id = 2;

COMMIT;  -- save changes permanently

-- Rollback on error
BEGIN;

UPDATE inventory SET stock = stock - 1 WHERE product_id = 42;
-- if something fails...
ROLLBACK; -- undo all changes in this transaction

-- SAVEPOINT — partial rollback
BEGIN;
INSERT INTO orders ...;
SAVEPOINT after_order;

INSERT INTO payments ...;
-- if payment fails, revert to after_order
ROLLBACK TO after_order;

COMMIT;
12 — Advanced

Indexes

Indexes speed up data retrieval by allowing the database to find rows without scanning the entire table. They come with a write-performance trade-off.

SQL — Index Management
-- Basic index on a single column
CREATE INDEX idx_emp_department
    ON employees (department);

-- Unique index (also enforces uniqueness)
CREATE UNIQUE INDEX idx_emp_email
    ON employees (email);

-- Composite (multi-column) index
CREATE INDEX idx_emp_dept_salary
    ON employees (department, salary);

-- Partial index (PostgreSQL) — index a subset of rows
CREATE INDEX idx_active_employees
    ON employees (name)
    WHERE is_active = TRUE;

-- Drop an index
DROP INDEX idx_emp_department;

-- View index usage (PostgreSQL)
SELECT indexname, indexdef
FROM pg_indexes
WHERE tablename = 'employees';
Best PracticesIndex columns used in WHERE, JOIN ON, and ORDER BY. Avoid over-indexing tables that are heavily written to.
13 — Advanced

Views

A view is a saved SELECT query that behaves like a virtual table. It doesn't store data itself (unless materialized).

SQL — Views
-- Create a view
CREATE VIEW active_employees AS
    SELECT id, name, department, salary
    FROM employees
    WHERE is_active = TRUE;

-- Query the view like a table
SELECT * FROM active_employees WHERE department = 'Engineering';

-- Replace/update a view
CREATE OR REPLACE VIEW active_employees AS
    SELECT id, name, department, salary, email
    FROM employees
    WHERE is_active = TRUE;

-- Materialized view (PostgreSQL) — stores result physically
CREATE MATERIALIZED VIEW dept_summary AS
    SELECT department, COUNT(*) headcount, AVG(salary) avg_sal
    FROM employees
    GROUP BY department;

-- Refresh materialized view
REFRESH MATERIALIZED VIEW dept_summary;

-- Drop a view
DROP VIEW active_employees;
14 — Advanced

Stored Procedures & Functions

SQL — Stored Procedure (MySQL)
DELIMITER $$

CREATE PROCEDURE give_raise(
    IN  dept       VARCHAR(50),
    IN  pct        DECIMAL(5,2),
    OUT rows_updated INT
)
BEGIN
    UPDATE employees
    SET    salary = salary * (1 + pct / 100)
    WHERE  department = dept;

    SET rows_updated = ROW_COUNT();
END$$

DELIMITER ;

-- Call the procedure
CALL give_raise('Engineering', 10, @updated);
SELECT @updated;
SQL — Function (PostgreSQL)
CREATE OR REPLACE FUNCTION get_dept_avg(dept_name TEXT)
RETURNS NUMERIC AS $$
BEGIN
    RETURN (
        SELECT AVG(salary)
        FROM   employees
        WHERE  department = dept_name
    );
END;
$$ LANGUAGE plpgsql;

-- Use in a query
SELECT get_dept_avg('Engineering');

Constraints

ConstraintDescription
PRIMARY KEYUniquely identifies each row; NOT NULL + UNIQUE
FOREIGN KEYLinks to primary key of another table
UNIQUEColumn values must be unique
NOT NULLColumn cannot store NULL
CHECKMust satisfy a boolean expression
DEFAULTDefault value when none is provided
SQL — Constraints in Practice
CREATE TABLE orders (
    id            INT           PRIMARY KEY AUTO_INCREMENT,
    customer_id   INT           NOT NULL,
    total_amount  DECIMAL(10,2) CHECK (total_amount >= 0),
    status        VARCHAR(20)  DEFAULT 'pending'
                                 CHECK (status IN ('pending','shipped','delivered')),
    order_date    TIMESTAMP     DEFAULT NOW(),

    CONSTRAINT fk_customer
        FOREIGN KEY (customer_id)
        REFERENCES customers(id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
16 — Advanced

Built-in Functions

SQL — String, Date & Conditional Functions
-- ── String Functions ──
SELECT
    UPPER(name),               -- 'JANE SMITH'
    LOWER(name),               -- 'jane smith'
    LENGTH(name),              -- character count
    TRIM(name),                -- remove leading/trailing spaces
    SUBSTRING(name, 1, 4),    -- 'Jane'
    REPLACE(email, '@old.com', '@new.com'),
    CONCAT(first_name, ' ', last_name),
    COALESCE(phone, 'N/A')    -- first non-NULL value
FROM employees;

-- ── Numeric Functions ──
SELECT
    ROUND(salary, 2),
    FLOOR(salary),
    CEIL(salary),
    ABS(balance),
    MOD(10, 3)                -- remainder = 1
FROM employees;

-- ── Date Functions ──
SELECT
    NOW(),                         -- current datetime
    CURDATE(),                     -- current date
    YEAR(hire_date),               -- extract year
    MONTH(hire_date),              -- extract month
    DATEDIFF(NOW(), hire_date),   -- days difference
    DATE_ADD(hire_date, INTERVAL 1 YEAR)
FROM employees;

-- ── Conditional: CASE WHEN ──
SELECT
    name,
    salary,
    CASE
        WHEN salary >= 100000 THEN 'Senior'
        WHEN salary >=  60000 THEN 'Mid'
        ELSE                        'Junior'
    END AS level
FROM employees;

-- NULLIF — returns NULL if two values are equal
SELECT NULLIF(score, 0) FROM results; -- avoid div-by-zero

-- IIF / IF shorthand (MySQL)
SELECT IF(is_active, 'Active', 'Inactive') FROM employees;
17 — Advanced

Normalization

Normalization is the process of organizing a database to reduce redundancy and improve data integrity.

Normal FormRule
1NFEach column holds atomic values; no repeating groups or arrays
2NF1NF + no partial dependencies (non-key columns depend on the whole primary key)
3NF2NF + no transitive dependencies (non-key columns depend only on the primary key)
BCNFStricter 3NF — every determinant must be a candidate key
4NFBCNF + no multi-valued dependencies
Practical AdviceMost production databases target 3NF. Intentional denormalization is sometimes done for read performance in reporting or analytics databases.
SQL — Quick Cheat Sheet
-- Clause execution order
-- FROM → JOIN → WHERE → GROUP BY → HAVING → SELECT → DISTINCT → ORDER BY → LIMIT

-- Find duplicates
SELECT email, COUNT(*) FROM employees
GROUP BY email HAVING COUNT(*) > 1;

-- Second highest salary
SELECT MAX(salary) FROM employees
WHERE salary < (SELECT MAX(salary) FROM employees);

-- Employees hired this year
SELECT * FROM employees
WHERE YEAR(hire_date) = YEAR(NOW());

-- Pivot with CASE WHEN
SELECT
    SUM(CASE WHEN department='Engineering' THEN 1 ELSE 0 END) eng,
    SUM(CASE WHEN department='Design'      THEN 1 ELSE 0 END) design,
    SUM(CASE WHEN department='Marketing'  THEN 1 ELSE 0 END) mkt
FROM employees;
